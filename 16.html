Introduction - Basic fibonacci problem (no scanner) import java.util.*; public class Main{ public static int fibDP(int x) { int fib[] = new int[x + 1]; fib[0] = 0; fib[1] = 1; for (int i = 2; i < x + 1; i++) { fib[i] = fib[i - 1] + fib[i - 2]; } return fib[x]; } public static void main(String[] args){ Scanner sc = new Scanner(System.in); int n = sc.nextInt(); System.out.println(fibDP(n)); } }
